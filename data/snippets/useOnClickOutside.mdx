title: "React useClickOutside hook"
publishedAt: "2019-02-09"
summary: "Maneja el evento click fuera del componente envuelto"
keywords: "Hooks React Javascript"
---

```js
import { useEffect } from "react";

export default function useOnClickOutside(ref, handler) {
  useEffect(() => {
    const listener = (event) => {
      if (!ref.current || ref.current.contains(event.target)) {
        return;
      }

      handler(event);
    };

    document.addEventListener("mousedown", listener);
    document.addEventListener("touchstart", listener);

    return () => {
      document.removeEventListener("mousedown", listener);
      document.removeEventListener("touchstart", listener);
    };
  }, [ref, handler]);
}
```

## Uso

```jsx
import { useRef } from "react";

const Menu = () => {
  const observerRef = useRef();
  
  useOnClickOutside(observerRef, () => alert("click outside"));

  return (
    <div ref={observerRef}>
      <p>Example, click out of this element</p>
    </div>
  );
};
```
