```js
// @ts-check

/** @type {string} */
let foo;

/** @type {string | number} */
let b;

/** @type {number[]} */
let w = [1, 2];

/** @type {{ name: string, age: number }} */
const person = {
  name: "Javier",
  age: 15,
};
```
